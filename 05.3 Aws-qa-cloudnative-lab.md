Crear una API demo, automatizar pruebas en AWS CodeBuild y CodePipeline, escanear vulnerabilidades con AWS Inspector y ejecutar una prueba móvil sencilla en AWS Device Farm.

### `src/server.js`

```javascript
const express = require("express");

const app = express();
app.use(express.json());

const students = [
  { id: 1, name: "Ana", course: "QA Cloud Native" },
  { id: 2, name: "Luis", course: "DevSecOps" }
];

app.get("/", (req, res) => {
  res.json({
    app: "AWS QA Cloud Native Lab",
    status: "ok"
  });
});

app.get("/health", (req, res) => {
  res.json({
    status: "healthy",
    timestamp: new Date().toISOString()
  });
});

app.get("/students", (req, res) => {
  res.json(students);
});

app.post("/students", (req, res) => {
  const { name, course } = req.body;

  if (!name || !course) {
    return res.status(400).json({
      error: "name and course are required"
    });
  }

  const newStudent = {
    id: students.length + 1,
    name,
    course
  };

  students.push(newStudent);
  res.status(201).json(newStudent);
});

if (require.main === module) {
  const port = process.env.PORT || 3000;
  app.listen(port, () => {
    console.log(`API running on port ${port}`);
  });
}

module.exports = app;
```

## 5. Pruebas automatizadas API

### `test/api.test.js`

```javascript
const request = require("supertest");
const app = require("../src/server");

describe("AWS QA Cloud Native API", () => {
  test("GET / should return app status", async () => {
    const response = await request(app).get("/");

    expect(response.statusCode).toBe(200);
    expect(response.body.status).toBe("ok");
  });

  test("GET /health should return healthy", async () => {
    const response = await request(app).get("/health");

    expect(response.statusCode).toBe(200);
    expect(response.body.status).toBe("healthy");
  });

  test("GET /students should return students", async () => {
    const response = await request(app).get("/students");

    expect(response.statusCode).toBe(200);
    expect(Array.isArray(response.body)).toBe(true);
  });

  test("POST /students should create student", async () => {
    const response = await request(app)
      .post("/students")
      .send({
        name: "María",
        course: "Automatización QA"
      });

    expect(response.statusCode).toBe(201);
    expect(response.body.name).toBe("María");
  });

  test("POST /students should validate required fields", async () => {
    const response = await request(app)
      .post("/students")
      .send({
        name: "Sin curso"
      });

    expect(response.statusCode).toBe(400);
    expect(response.body.error).toBe("name and course are required");
  });
});
```

## 8. AWS CodeBuild

### ¿Qué hará?

CodeBuild ejecutará el siguiente flujo:

```
Clonar código
  → Instalar dependencias
  → Ejecutar pruebas
  → Generar cobertura
  → Construir imagen Docker
```

### `Agrega el contenido del archivo buildspec.yml`

```yaml
version: 0.2

env:
  variables:
    IMAGE_NAME: "aws-qa-cloudnative-lab"

phases:
  install:
    runtime-versions:
      nodejs: 20
    commands:
      - echo "Instalando dependencias..."
      - npm ci

  pre_build:
    commands:
      - echo "Ejecutando pruebas automatizadas..."
      - npm test

  build:
    commands:
      - echo "Construyendo imagen Docker..."
      - docker build -t $IMAGE_NAME:latest .

  post_build:
    commands:
      - echo "Build finalizado correctamente."
      - echo "Fecha:"
      - date

artifacts:
  files:
    - '**/*'
```

### `Sube los cambios a Git buildspec.yml`
```yaml
git add buildspec.yml
git commit -m "Fix CodeBuild buildspec"
git push
```

## 9. Configuración de AWS CodeBuild

### Objetivo

Configurar un proyecto de AWS CodeBuild para ejecutar pruebas automatizadas y construir imágenes Docker a partir de un repositorio GitHub.

---

### 9.1 Configuración del proyecto

**Nombre del proyecto:**

```text
qa-cloudnative-codebuild-<nombre-alumno>
```

Ejemplo:

```text
qa-cloudnative-codebuild-jorge
```

**Tipo de proyecto:**

```text
Proyecto predeterminado
```

**Configuración adicional** — dejar todos los valores por defecto. No habilitar:

- Acceso público a la compilación
- Insignia de compilación
- Límite de compilación simultánea

---

### 9.2 Origen

| Campo | Valor |
|---|---|
| Proveedor del origen | GitHub |
| Tipo de repositorio | Repositorio público |
| Versión de origen | `main` |

**URL del repositorio** (ejemplo):

```text
https://github.com/Automatizacion-Pruebas-de-Software/aws-qa-cloudnative-lab2.git
```

**Configuración adicional:**

- Profundidad del clon Git: `1`
- Submódulos Git: no habilitar

---

### 9.3 Entorno

| Campo | Valor |
|---|---|
| Modelo de aprovisionamiento | Bajo demanda |
| Imagen del entorno | Imagen administrada |
| Computación | EC2 |
| Modo de ejecución | Contenedor |
| Sistema operativo | Amazon Linux |
| Tiempo de ejecución | Standard |
| Imagen | `aws/codebuild/amazonlinux-x86_64-standard:6.0` |
| Versión de la imagen | Utilizar siempre la imagen más reciente |
| Computación acelerada por GPU | No habilitar |

---

### 9.4 Rol de servicio

Seleccionar **Nuevo rol de servicio** y utilizar el nombre generado automáticamente por AWS.

Ejemplo:

```text
codebuild-qa-cloudnative-codebuild-jorge-service-role
```

---

### 9.5 Configuración adicional del entorno

| Campo | Valor |
|---|---|
| Reintentos automáticos | `0` |
| Tiempo de espera | 0 horas, 15 minutos |
| Tiempo de espera en cola | 8 horas, 0 minutos |
| Certificado | No instalar ningún certificado |
| VPC | No configurar |
| Tipo de computación | 3 GB memoria, 2 vCPU |
| Docker Server | No habilitar |
| Credencial de registro | Dejar vacío |

```text

```

**Detección automática de informes** — mantener habilitada (no marcar "Desactivar la detección automática de informes").

### Resultado esperado

Al finalizar la configuración, AWS CodeBuild será capaz de:

1. Clonar automáticamente el repositorio GitHub.
2. Leer el archivo `buildspec.yml`.
3. Instalar dependencias con `npm ci`.
4. Ejecutar pruebas automatizadas con `npm test`.
5. Construir imágenes Docker mediante `docker build`.
6. Generar registros en CloudWatch.
7. Servir como base para la integración posterior con:
   - AWS CodePipeline
   - Amazon ECR
   - AWS Inspector
   - AWS Device Farm

Este laboratorio representa el primer bloque de una arquitectura moderna de QA Cloud-Native basada en integración continua y automatización de pruebas en AWS.
